🖱️ Actividad 2: Arrastrar y Soltar - Clasifica Sentencias DML

Arrastra cada sentencia SQL a la operación CRUD que le corresponde (CREATE, READ, UPDATE o DELETE).

INSERT INTO usuarios (nombre, email) VALUES ('Juan', 'juan@email.com')
SELECT nombre, email FROM usuarios WHERE estado='activo'
UPDATE usuarios SET email='nuevo@email.com' WHERE id=1
DELETE FROM usuarios WHERE id=5
INSERT INTO productos (nombre, precio) VALUES ('Laptop', 800)
SELECT * FROM pedidos ORDER BY fecha DESC
UPDATE productos SET precio=950 WHERE id_producto=3
DELETE FROM logs WHERE fecha < '2024-01-01'
